#Table course definition :
CREATE TABLE course
    -> (
    -> courseCode varchar(10) PRIMARY KEY,
    -> courseName varchar(64) NOT NULL,
    -> branch varchar(64),
    -> sem int,
    -> author varchar(64),
    -> description text,
    -> file varchar(128),
    -> dateCreated Date);

#Table for branches
CREATE TABLE branch
    -> (
    -> branchName varchar(64) PRIMARY KEY,
    -> courseSem1 text,
    -> courseSem2 text,
    -> courseSem3 text,
    -> courseSem4 text,
    -> courseSem5 text,
    -> courseSem6 text,
    -> courseSem7 text,
    -> courseSem8 text);

#Table for admins
create table admins
    -> (
    -> id int(5) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    -> username VARCHAR(64) NOT NULL,
    -> hashedPassword varchar(64) NOT NULL);
